# QUESTÃO 3

A questão pede para que a gente crie uma função em assembly para 
alterar alguns valores de um array. Para isso comecei criando os 
registradores que armazenam o iterador e a soma, depois iniciei um 
label loop onde teria a comparação do iterador com o n passado para 
a função, que representa o tamanho do vetor e se eles forem iguais 
significa que chegou no final do vetor e pula para fora do loop. 
Nesse loop, ele adiciona o valor do elemento atual na soma, que 
será usada para alterar o valor do array caso ele seja 0, entao 
comparo o elemento com 0, se for 0 eu troco pelo valor da soma e 
zero a soma, se nao eu so continuo o loop

    .text
    .globl foo
    foo:
        /* PROLOGO */
        pushq %rbp            
        movq  %rsp, %rbp      
        subq  $16, %rsp       
        movq  %rbx, -8(%rbp)  
        movq  %r12, -16(%rbp) 

        movl $0, %ebx        /* int i = 0 */
        movl $0, %ecx        /* int s = 0 */

    loop:
        cmpl %ebx, %esi
        je fora

        addl (%rdi), %ecx
        cmpl $0, (%rdi)
        jne else

        movl %ecx, (%rdi)
        movl $0, %ecx

    else:
        addl $1, %ebx
        addq $4, %rdi
        jmp loop

    fora:
        leave
        ret
